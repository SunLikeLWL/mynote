### 1、翻转链表

function reverse(nodeList){
    var curNode = nodeList;
    var prevNode = null;
    var nextNode = null;
    while(curNode){
        nextNode = curNode.next;
        curNode.next = prevNode;
        prevNode = curNode; 
        curNode = nextNode;
    }
}

### 给一个String，判断大、中、小括号是否合法

方法1：堆栈
O(n)

function isValid(s) {
   
    let stack = [];
    map = { ')': '(', ']': '[', '}': '{' };
    for (let i = 0; i < s.length; i++) {
        console.log(s[i],(s[i] in map))
        if (!(s[i] in map)) {
            stack.push(s[i])
        }
        else {
            if (map[s[i]] !== stack.pop()) {
                return false;
            }

        }
    }
    console.log(stack)
    if (stack.length===0) {
        return true
    }
    return false;
}

console.log(isValid('([{])'));//false
console.log(isValid('([{}])'))；//true



方法2：replace
O(n^2)

function isValid(s){
    let l;
    do{
         l = s.length;
         s = s.replace("()","").replace("[]","").replace("{}","");
    }while(l!=s.length)
    return s.length ===0;
}

console.log(isValid("{[()]}"));//true
console.log(isValid("{[([)]}"));//false
